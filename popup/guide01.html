# Javascript

	<script type="text/javascript" src="http://static.ssgcdn.com/ui/ssg/js/common/hivetab_min.js"></script>
	<script type="text/javascript">
		$(function(){
			$('#tabType').hivetab();
		});
	</script>

# 옵션

	// 자동롤링
	$(function(){
		$('#tabType').hivetab({
			auto:'random'
			//값 : auto, random
		});
	});

	// 클릭
	$(function(){
		$('#tabType').hivetab({
			event : 'click'
		});
	});

	// tp_tab 클릭시 경로 이동
	$(function(){
		$('#tabType').hivetab();
		$('.tp_tab').on('click', function() {
			window.location = $(this).attr('href');
		})
	});

	// 자동롤링 시간 설정
	$(function(){
		$('#tabType').hivetab({
			auto : 'auto',
			play_time : 5000
		});
	});
